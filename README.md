## Install

```bash
npm install --save react-js-dropdavn
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-js-dropdavn'
import 'react-js-dropdavn/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
